create table userdetails (
	id int primary key,
	username varchar(50) unique not null,
	username varchar(50) not null,
	userpassword varchar(50) not null,
	accountnonexpired bool default true,
	accountnonlocked bool default true,
	credentialsnonexpired bool default true,
	enabled bool default true
);

create table roles (
	id int primary key,
	name varchar (15) not null
);

create table user_role(
	user_id int,
	role_id int,
	constraint userid_userdetails foreign key (user_id) references userdetails (id),
	constraint id_roles foreign key (role_id) references roles (id),
	constraint primarykey primary key(user_id, role_id)
);

	unique_id varchar(20) primary key,
	product_id int not null unique,
	size varchar(3),
	dominant_material varchar(50),
	actual_color varchar(50),
	dominant_color varchar(50),
	product_details varchar(2000),
